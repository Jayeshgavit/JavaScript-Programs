Error Handling

Example with try and catch block : 

            try{
                rs = 5 / 0;
                console.log("result is :", rs);
            }
            catch(error){
                consolelog('An error occurred : ', error.message);
            }

Throw Error custom : throw

            function checkAge(age){
                if(age < 18){
                    throw new Error(" you must be 18 years old");
        
                }
                return 'access granted';
            }
        
            try{
                let result = checkAge(12);
                console.log(result);
            }catch(err){
                console.log(err.message);
            }

finally Block : 

            try{
                let str = 'Hello! Bro!!';
                console.log(str.toUpperCase());
        
            }
            catch(err1){
