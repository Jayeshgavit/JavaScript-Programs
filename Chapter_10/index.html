execute multiple callback function with delayes :

You can get multiple data with delayes (ex-1 ---> 2sec ex-2 --> ex-3 2sec)

            function getData(dataId,getNextData){
                setTimeout(()=>{
                    console.log('data ',dataId);
                    getNextData();
                },9000);
            }
            // getData(257, ()=>{
            //     getData(512);
            // });
            
            //multiple delayes
            getData(1, ()=>{
                getData(2,()=>{
                    getData(3, ()=>{
                        getData(4);
                    })
                });
            });
